    Velociadade do carro: <input type="number" name= "txtvel" id= "txtvel">Km/h <input type="button" value="verificar" onclick="verificar()">

    <div id="res"></div>

    <script>
        function verificar() {
            var txtvel = window.document.querySelector ('input#txtvel')
            var res = document.querySelector ('div#res')
            var vel = Number(txtvel.value)
            res.innerHTML = `Sua velocidade é <strong>${vel}Km/h</strong>`
            if (vel > 60) {
                res.innerHTML += ' Você ultrapassou o limite de velocidade. <STRONG>MULTADO!</STRONG>'
            }
                }
